var wpcom_remote_login_extra_auth = '';
function wpcom_remote_login_remove_dom_node_id( element_id ) {
	var dom_node = document.getElementById( element_id );
	if ( dom_node ) { dom_node.remove(); }
}
function wpcom_remote_login_remove_dom_node_classes( class_name ) {
	var dom_nodes = document.querySelectorAll( '.' + class_name );
	for ( var i = 0; i < dom_nodes.length; i++ ) {
		dom_nodes[i].remove();
	}
}
function wpcom_remote_login_final_cleanup() {
	wpcom_remote_login_remove_dom_node_classes( "wpcom_remote_login_msg" );
	wpcom_remote_login_remove_dom_node_id( "wpcom_remote_login_key" );
	wpcom_remote_login_remove_dom_node_id( "wpcom_remote_login_validate" );
	wpcom_remote_login_remove_dom_node_id( "wpcom_remote_login_js" );
	wpcom_remote_login_remove_dom_node_id( "wpcom_request_access_iframe" );
	wpcom_remote_login_remove_dom_node_id( "wpcom_request_access_styles" );
}

// Watch for messages back from the remote login
window.addEventListener( "message", function( e ) {
	if ( e.origin === "https://r-login.wordpress.com" ) {
		var data = {};
		try {
			data = JSON.parse( e.data );
		} catch( e ) {
			wpcom_remote_login_final_cleanup();
			return;
		}

		if ( data.msg === 'LOGIN' ) {
			// Clean up the login check iframe
			wpcom_remote_login_remove_dom_node_id( "wpcom_remote_login_key" );

			var id_regex = new RegExp( /^[0-9]+$/ );
			var token_regex = new RegExp( /^.*|.*|.*$/ );
			if (
				token_regex.test( data.token )
				&& id_regex.test( data.wpcomid )
			) {
				// We have everything we need to ask for a login
				var script = document.createElement( "script" );
				script.setAttribute( "id", "wpcom_remote_login_validate" );
				script.src = '/remote-login.php?wpcom_remote_login=validate'
					+ '&wpcomid=' + data.wpcomid
					+ '&token=' + encodeURIComponent( data.token )
					+ '&host=' + window.location.protocol
						+ '//' + window.location.hostname;
				document.body.appendChild( script );
			}

			return;
		}

		// Safari ITP, not logged in, so redirect
		if ( data.msg === 'LOGIN-REDIRECT' ) {
			window.location = 'https://wordpress.com/log-in?redirect_to=' + window.location.href;
			return;
		}

		// Safari ITP, storage access failed, remove the request
		if ( data.msg === 'LOGIN-REMOVE' ) {
			var css_zap = 'html { -webkit-transition: margin-top 1s; transition: margin-top 1s; } /* 9001 */ html { margin-top: 0 !important; } * html body { margin-top: 0 !important; } @media screen and ( max-width: 782px ) { html { margin-top: 0 !important; } * html body { margin-top: 0 !important; } }';
			var style_zap = document.createElement( 'style' );
			style_zap.type = 'text/css';
			style_zap.appendChild( document.createTextNode( css_zap ) );
			document.body.appendChild( style_zap );

			var e = document.getElementById( 'wpcom_request_access_iframe' );
			e.parentNode.removeChild( e );

			document.cookie = 'wordpress_com_login_access=denied; path=/; max-age=31536000';

			return;
		}

		// Safari ITP
		if ( data.msg === 'REQUEST_ACCESS' ) {
			console.log( 'request access: safari' );

			// Check ITP iframe enable/disable knob
			if ( wpcom_remote_login_extra_auth !== 'safari_itp_iframe' ) {
				return;
			}

			// If we are in a "private window" there is no ITP.
			var private_window = false;
			try {
				var opendb = window.openDatabase( null, null, null, null );
			} catch( e ) {
				private_window = true;
			}

			if ( private_window ) {
				console.log( 'private window' );
				return;
			}

			var iframe = document.createElement( 'iframe' );
			iframe.id = 'wpcom_request_access_iframe';
			iframe.setAttribute( 'scrolling', 'no' );
			iframe.setAttribute( 'sandbox', 'allow-storage-access-by-user-activation allow-scripts allow-same-origin allow-top-navigation-by-user-activation' );
			iframe.src = 'https://r-login.wordpress.com/remote-login.php?wpcom_remote_login=request_access&origin=' + encodeURIComponent( data.origin ) + '&wpcomid=' + encodeURIComponent( data.wpcomid );

			var css = 'html { -webkit-transition: margin-top 1s; transition: margin-top 1s; } /* 9001 */ html { margin-top: 46px !important; } * html body { margin-top: 46px !important; } @media screen and ( max-width: 660px ) { html { margin-top: 71px !important; } * html body { margin-top: 71px !important; } #wpcom_request_access_iframe { display: block; height: 71px !important; } } #wpcom_request_access_iframe { border: 0px; height: 46px; position: fixed; top: 0; left: 0; width: 100%; min-width: 100%; z-index: 99999; background: #23282d; } ';

			var style = document.createElement( 'style' );
			style.type = 'text/css';
			style.id = 'wpcom_request_access_styles';
			style.appendChild( document.createTextNode( css ) );
			document.body.appendChild( style );

			document.body.appendChild( iframe );
		}

		if ( data.msg === 'DONE' ) {
			wpcom_remote_login_final_cleanup();
		}
	}
}, false );

		<h1 class="entry-title">2.17</h1>

				<div class="entry-meta">
						<span class="sep">Posted on </span><a href="https://wanderinginn.com/2017/04/22/2-16/" title="10:52 pm" rel="bookmark"><time class="entry-date" datetime="2017-04-22T22:52:37+00:00">April 22, 2017</time></a><span class="by-author"> <span class="sep"> by </span> <span class="author vcard"><a class="url fn n" href="https://wanderinginn.com/author/pirateaba/" title="View all posts by pirateaba" rel="author">pirateaba</a></span></span>		</div><!-- .entry-meta -->

	<div class="entry-content">
		<p><em>Ryoka stumbled into the cave, bleeding. Dying. Every step was agony. She felt her sliced flesh and fragments of rock embedded in her skin cutting away at her sanity.</em></p>
<p><em>Pain. She didn’t know why she was still moving. But as Ryoka stumbled forwards, the rough stone turned to marble. The room filled with treasures made of gold and magic. </em></p>
<p><em>She left footsteps traced in crimson as she walked forwards, forgetting her hurts for a second. The wonders around her were far more glorious than anything she’d ever seen in a museum. Artwork as beautiful as a Louvre masterpiece hung next to swords that glimmered with magic. Yes, it was in the air.</em></p>
<p><em>Magic.</em></p>
<p><em>Ryoka felt her heart beating faintly in her chest. She heard rushing wind, but then the cavern wall opened up and her heart did stop.</em></p>
<p><em>Because in the center of this cave of miracles was a Dragon.</em></p>
<p><em>A Dragon. A massive creature four times the size of a house, slumbering on the cold marble. A creature straight out of myth and dreams. </em></p>
<p><em>Serpent. Wyrm. Drake. Arach. Naga. Ormr. Tanniym. Vovin. Draak. Drage. Draeke.</em></p>
<p>Dragon.</p>
<p><em>The massive dragon was sleeping in an open space in the center of the cavern. Ryoka stopped. She stared. His scales </em>shone<em> as the light touched them, so brilliant that her heart ached.</em></p>
<p><em>Her eyes filled with tears. Ryoka sank to her knees as her blood covered the marble floor. She didn’t care. It was worth it. This was worth it.</em></p>
<p><em>It was worth seeing this before she died.</em></p>
<p><em>“Hm. Who’s there?”</em></p>
<p><em>His head moved. The dragon opened both eyes and stared down at Ryoka. She gazed into two massive pupils, mismatched colors of heliotrope and cerulean. Celestial lights shimmered within the moving pupils.</em></p>
<p><em>For a moment, Ryoka understood magic. She gazed into the center of it; the reason why it could exist. A fragment of the truth. A piece of forever. A dragon’s eye.</em></p>
<p><em>Her eyes rolled back up in her head. Ryoka fell forwards and her nose broke as she toppled onto the floor.</em></p>
<p><em>The Dragon grumbled deep in his throat. He looked around, and </em>tsked<em> at the blood trail.</em></p>
<p><em>“I’m going to have to clean that up as well, aren’t I? And fix the nose. And use a memory spell. Honestly, this is why humans are…”</em></p>
<p><em> </em></p>
<p><em>&#8212;-</em></p>
<p><em> </em></p>
<p>She remembered.</p>
<p><em>“Awaken, human.”</em></p>
<p><em>Ryoka opened her eyes. She sat up at once and felt no pain. She was healed? Healed! How?</em></p>
<p><em>It was a miracle. But then Ryoka noticed the old man standing in front of her. Long white robes covered a body only lightly touched by age. His mismatched eyes gazed down at her as he gestured at her.</em></p>
<p><em>“Rise, human.”</em></p>
<p><em>Ryoka stood up. She couldn’t help it. She gaped at the old man who had pointed ears and flawless features.</em></p>
<p><em>“Mm. You must be exhausted from your travels. I confess; it was unusual for me to find one in such bad shape. But I have healed your wounds. I presume you are the Courier I requested?”</em></p>
<p><em>Ryoka blinked. She stared open-mouthed at the old man. He frowned at her.</em></p>
<p><em>“Hello? Can you hear me?”</em></p>
<p><em>The human girl made no reply. Teriarch looked around, frowning.</em></p>
<p><em>“What? What is—”</em></p>
<p><em>He turned his head towards the cavern wall and paused.</em></p>
<p><em>“Ah.”</em></p>
<p><em>Teriarch’s illusory human form was perfect. He looked exactly like a normal human, albeit one made in flawless perfection and with slightly elven features. But he had forgotten one important aspect.</em></p>
<p><em>Ryoka stared at the shadow of the dragon looming huge on the wall behind Teriarch. It stretched out from his robes, lengthening impossibly until it perfectly reflected the dragon, not the illusion.</em></p>
<p><em>“My god. What—”</em></p>
<p><em>Teriarch sighed. He pointed at Ryoka, and she felt her head go blank.</em></p>
<p><em>“Let’s try that again.”</em></p>
<p><em> </em></p>
<p><em>&#8212;-</em></p>
<p><em> </em></p>
<p>Memory flooded into her. Words.</p>
<p><em>Teriarch stared hard at Ryoka as she stood, immobilized by the magic on her. He frowned, tapped at a lip.</em></p>
<p><em>“What has happened in the world since I have been asleep?”</em></p>
<p><em>Ryoka blinked at him. Since he was asleep?</em></p>
<p><em>“How long have you been asleep?”</em></p>
<p><em>“Ah. Well, I mean…”</em></p>
<p><em>Teriarch blinked at Ryoka a few times, and then waved his hand in front of her face.</em></p>
<p><em>“Forget that.”</em></p>
<p><em>Ryoka blinked. Teriarch frowned at her. He paused, and then spoke.</em></p>
<p><em>“I wish to know of any ongoing major wars between multiple nations, new technologies or spells developed, legendary monsters sighted or slain—import of that nature. Tell me all the news you have heard of in the past several years or so. What has transpired while I have been…secluded here?”</em></p>
<p><em>The human girl paused, and then shrugged. Even bound by the truthfulness spell, her tone projected quite a lot of insolence.</em></p>
<p><em>“Dunno.”</em></p>
<p><em>“Excuse me?”</em></p>
<p><em>“I don’t pay attention to world news. You probably know more about what’s going than I do.”</em></p>
<p><em>Teriarch harrumphed and glared at Ryoka, but the truth spell on her was still active. He changed tact.</em></p>
<p><em>“Do you at least know if Magnolia Reinhart still lives?”</em></p>
<p><em>“…Yes.”</em></p>
<p><em>He nods sagely, stroking his beard.</em></p>
<p><em>“Indeed, indeed. That is good. Well then, have you heard of a strange child known as Ryoka Griffin? She is a Runner, like you, although presumably high-level in the [Runner] class, unlike you.”</em></p>
<p><em>Ryoka opened her mouth and Teriarch continued, irritated.</em></p>
<p><em>“Although why Reinhart wants me to find the wretched girl, I can’t imagine. Do I </em>look<em> like I have the time to search for one annoying bratling all the time?”</em></p>
<p><em>“…Didn’t you just say you didn’t know if Magnolia was alive?”</em></p>
<p><em>Teriarch paused and then pointed at Ryoka.</em></p>
<p><em>“Forget that too. I do not know Magnolia Reinhart.”</em></p>
<p><em>He paused, and then amended his tone. Teriarch looked at Ryoka.</em></p>
<p><em>“Now, tell me where Ryoka Griffin is. Reinhart said—blast!”</em></p>
<p><em>He poked Ryoka in the head again and she forgot. Teriarch closed his eyes.</em></p>
<p><em>“This is why…”</em></p>
<p><em> </em></p>
<p><em>&#8212;-</em></p>
<p><em> </em></p>
<p><em>Ryoka paused as she stared at a massive broadsword hanging from one of the walls of the cavern. It was far too large for any human to use, or even a Minotaur. Teriarch stroked his beard proudly.</em></p>
<p><em>“Ah. Are you admiring my collection? That was a giant’s weapon, once. I took it from him when I—well, it was another time. Note the exquisite tempering of the blade.”</em></p>
<p><em>“All this gold and magical items…”</em></p>
<p><em>Ryoka stared around at the room, freed from Teriarch’s spell. She looked at a pile of ruby gemstones, each one larger than her fist.</em></p>
<p><em>“I’m…a collector.”</em></p>
<p><em>“This isn’t a collection. This looks…like a hoard?”</em></p>
<p><em>Teriarch waved his hands in Ryoka’s face hurriedly. </em></p>
<p><em>“This is </em>not<em> a hoard. You will not think it is a hoard.”</em></p>
<p><em>Ryoka blinked emptily at the air for a few seconds. Teriarch paused. Then he spoke conversationally.</em></p>
<p><em>“Do you like my collection? I have amassed it from every corner of the world.”</em></p>
<p><em>She gazed around, taking in the magnificence of the room for the first time. Ryoka had to admit, it was impressive beyond belief.</em></p>
<p><em>“How did you collect it all here? It looks…magical.”</em></p>
<p><em>“Oh, it’s quite magical. But I have my ways. And besides, when you can fly, amassing such things is fairly simple.”</em></p>
<p><em>Ryoka stared at Teriarch. He paused, and swallowed.</em></p>
<p><em>“Flying with magic, is what I mean. Not that I know of any other ways to fly, of course.”</em></p>
<p><em>Her eyes narrowed slightly.</em></p>
<p><em>“Right.”</em></p>
<p><em>“Of course. Ahem. Do note the Orb of Scrying. It’s embedded in pure jade and enchanted with several enhancing spells that increase the range and scope of the spell itself.”</em></p>
<p><em>Ryoka glanced into it and froze. Teriarch stared at her with concern.</em></p>
<p><em>“What? Did I leave it on?”</em></p>
<p><em>He looked over and cursed.</em></p>
<p><em>“Tamaroth’s beard!”</em></p>
<p><em>The Scrying Orb was reflecting the room they were standing in. It showed a small human girl standing right in front of a golden dragon. This time Teriarch’s expletive made the cavern shake.</em></p>
<p><em>Teriarch thrust one claw at Ryoka as she turned to run. He spoke curtly.</em></p>
<p><em>“That’s it. Forget everything except the part where I gave you your instructions. Get out. Come back in two minutes.”</em></p>
<p><em>Ryoka turned around and walked out of the cavern. Teriarch cast an irritated glance towards the magical orb and his tail </em>flicked<em> the globe off its pedestal. It shattered in an explosion of mist and magic against one wall.</em></p>
<p><em>“Wretched thing.”</em></p>
<p><em> </em></p>
<p><em>&#8212;-</em></p>
<p><em> </em></p>
<p>And once more. One last memory made Ryoka shudder on the hardwood floor of the inn.</p>
<p><em> </em></p>
<p><em>The stink potion was keeping all the monsters away. Ryoka had to smile as she walked towards the entrance of Teriarch’s cave. But she froze. Down the winding canyon path, she saw something out of the corner of her eye.</em></p>
<p><em>It was just for a second. But Ryoka saw it duck back behind a large boulder as she whirled. She froze in place, and her hand went to her side. She had a knife she used for camping and other tasks at her belt. Normally she’d fight with her hands, but…</em></p>
<p><em>The thing peeked out its head again, and Ryoka’s hand gripped the hilt so hard her knuckles </em>cracked.</p>
<p><em>It looked vaguely human. It had a human head, a normal torso and legs that protruded slightly around the edge of the boulder. But it was naked.</em></p>
<p><em>And it wasn’t </em>right.</p>
<p><em>Its face wasn’t right. One of its eyes twisted right until it was nearly vertical, and the other one was looking in the opposite direction of the first. </em></p>
<p><em>A tiny pupil stared at Ryoka from across the hundred feet that separated them. Then the thing came out from behind the boulder, and Ryoka froze. Her mind was overwhelmed by horror. It shouldn’t move like that. It shouldn’t—</em></p>
<p><em>It reached out for her, one horrible clawed arm stretching out, out across the long distance. Flesh stretched like gum, skin rippling and peeling to reveal something red and pulsating inside. Ryoka couldn’t move. Her body was filled with true terror. The thing’s finger reached out for her face—</em></p>
<p><em>And a </em>roar <em>of fury filled the air. The creature looked up, and suddenly its hand retracted in an instant. It turned and ran as a massive shadow turned the world around Ryoka dark. She looked up.</em></p>
<p><em>Teriarch flew overhead, golden scales lighting up even parts of the mountain as the sunlight bounced off them. He landed with a </em>crash<em> on the ground that sent Ryoka to her knees. She couldn’t do anything but stare at him.</em></p>
<p><em>Teriarch’s massive head snapped towards the thing and it fled, running in a jerking, halting manner back around a bend in the canyon. He breathed fire after it—a concentrated jet of flames that missed his target by inches. Then the creature was gone.</em></p>
<p><em>The dragon—Teriarch—sighed and looked around at Ryoka. The her of then had no idea who he was, and she could only stare up at him, transfixed. His voice was like an avalanche heard from far away as he spoke to her.</em></p>
<p><em>“You must be warier, Ryoka Griffin. Darker things lurk in the High Passes. They do not smell—or if they do, they have different opinions than we do.” </em></p>
<p><em>He shrugged, as if the encounter meant nothing. To him, it did not.</em></p>
<p><em>“Come in, then. We have much to discuss. Forget what you have seen here. It will only burden you.”</em></p>
<p><em>Then and only then did the Frost Faeries return, laughing, and the Dragon flew back into his cave, and breathed fire. The memory of the&#8230;thing&#8230;vanished from Ryoka&#8217;s mind as she went to confront Teriarch.</em></p>
<p>She had forgotten. But now she remembered. Ryoka remembered it. She knew. And the knowledge was glorious. She had seen one. Seen a legend in the flesh.</p>
<p>A <em>dragon.</em></p>
<p>&nbsp;</p>
<p>&#8212;-</p>
<p>&nbsp;</p>
<p>Erin stared worriedly at Ryoka as the young woman lay on the floor, staring blankly up at the ceiling. She looked around at Val and the others.</p>
<p>“Shouldn’t we do something? It’s been thirty minutes already.”</p>
<p>Ceria shook her head. She was sitting in a chair, watching Ryoka intently. Other than moving Ryoka so she wasn’t lying at an awkward angle, she hadn’t let Erin or anyone else touch her.</p>
<p>“It’s best not to, especially if she’s experiencing something magically. Movement can mess with her perceptions and affect her mind. Unless you’ve got a different opinion, Pisces? You know more mind manipulation spells than I do.”</p>
<p>Erin looked at Pisces. He was inspecting Ryoka just as intently. She remembered that he <em>did</em> know some spells that created illusions, like when she’d first met him.</p>
<p>But he just shook his head absently.</p>
<p>“This is far beyond my abilities to comprehend. There was a warding spell mixed into the first spell, did you notice?”</p>
<p>“<em>I </em>did.”</p>
<p>Olesm said that from a far table. Val nodded. He was crouched next to Ryoka, studying her face.</p>
<p>“So did I. That spell took out a powerful charm. I need to see a [Mage] about that. I was told my charm would catch any spell and remove any enchantment.”</p>
<p>“Now <em>there’s</em> an empty boast.”</p>
<p>Ceria’s words were the last thing spoken for a while. Everyone watched Ryoka sleeping—or dreaming—in silence.</p>
<p>Erin worried. She wasn’t sure if she and Ryoka were good friends—they’d barely spent more than a day together, although they had rescued Ceria and fought monsters, so that had to count. But she cared about Ryoka. Not only was she Erin’s only lifeline to her world, but Ryoka knew how to do things Erin did not.</p>
<p>It seemed like an hour had passed, but probably only a few minutes had when Ryoka suddenly sat up, gasping and clutching at her head.</p>
<p>“Ryoka!”</p>
<p>Erin leapt forwards, but Val was closer. He steadied Ryoka with one hand as she flailed and grabbed at his arm.</p>
<p>“Easy. Ryoka, are you okay? Did the charm work?”</p>
<p>“What? What? I—I’m okay. Yes. It worked.”</p>
<p>Ryoka stared around the room, wide-eyed and out of breath. Ceria abandoned her seat to come over, and Olesm moved one table closer.</p>
<p>“What did you see? Did you remember anything important?”</p>
<p>“I—it was—”</p>
<p>Ryoka closed her eyes. When she opened them, something flickered behind her gaze as she shook her head. Erin saw a tinge of sadness—joy, even a bit of fear, but most of all, what she could only say was wonder in that look. It was an expression she’d never seen Ryoka make.</p>
<p>It looked…innocent. Not like Ryoka’s normally unreadable expression or her dour face.</p>
<p>“It’s a secret. I’m sorry. But I can’t say.”</p>
<p>“Damn. I thought as much.”</p>
<p>Val shook his head. He was picking up fragments of the burnt dreamcatcher off the floor. Erin really, <em>really</em> doubted he could fix them, but maybe they were still useful? She began to help him pick them up.</p>
<p>“I’m uh, I’m sorry about your charm.”</p>
<p>Ryoka stared at the pieces on the floor, and then her singed clothing. Val sighed as he nodded and Erin looked at him.</p>
<p>“Was it expensive?”</p>
<p>“You could say that. I paid over a thousand gold coins for that charm.”</p>
<p>“A <em>thousand—</em>”</p>
<p>Pisces <em>choked</em> and spat out a mouthful of water. Ceria stared at the blackened remains of the dreamcatcher and Ryoka turned pale.</p>
<p>“I can pay you back for it. I’ve got a delivery—”</p>
<p>Val waved a hand at her, shushing Ryoka. He looked at her seriously.</p>
<p>“Just tell me, from one Runner to another. Was it worth it?”</p>
<p>Ryoka hesitated. Then she nodded fervently.</p>
<p>“It was. It was worth anything and everything.”</p>
<p>“Then forget about the cost. It was my decision to lend it to you—I’ll take it as a lesson on giving things away. Although whether it was a mistake or not…”</p>
<p>Val shrugged. Then he grinned.</p>
<p>“Today is full of surprises, huh?”</p>
<p>Pisces and Ceria gaped at Val, and you could have picked Olesm’s jaw off the floor, but Ryoka only smiled. She flexed her hand and stared past it. She smiled, with such genuine feeling that Erin was taken aback.</p>
<p>“Thank you. It <em>was</em> worth it.”</p>
<p>“Think nothing of it. We’ve got to look out for each other, right?”</p>
<p>A thousand gold coin’s worth of magic didn’t sound like ‘nothing’ to Erin, but Val just swept the few burnt pieces into a bag and put it in his belt pouch. And that was it. Debt forgiven. That was so awesome Erin didn’t have words for it.</p>
<p>Val was a stand-up guy. Erin smiled, ran into the kitchen, and came out with a steaming burger which she put in front of him.</p>
<p>“I can’t pay for that, but have another hamburger on the house.”</p>
<p>It had been meant for Ryoka, but the other girl was still staring at nothing, and Erin didn’t think she’d mind waiting a few more minutes. Val eyed the hamburger. He’d already eaten four, but his stomach rumbled. He smiled up at Erin.</p>
<p>“You are very kind. But ah, could I have some of that red stuff on it? What did you call it?”</p>
<p>“Ketchup. But that’s extra. Pay up.”</p>
<p>Val blinked at Erin. She grinned.</p>
<p>“Kidding!”</p>
<p>&nbsp;</p>
<p>&#8212;-</p>
<p>&nbsp;</p>
<p>Ryoka was still trying to get her head back together when Erin came back with her food. Absentmindedly, Ryoka raised the burger to her lips and bit.</p>
<p>Perhaps it was a testament to nostalgia, Erin’s improving skill at cooking, or Ryoka’s state of starvation, but <em>that</em> was what brought her back into reality. Yes. A well-made hamburger could erase even the wonder of meeting a dragon.</p>
<p>At least for ten seconds, which was roughly how long it took Ryoka to inhale the burger. Erin brought her another, and Ryoka repeated the trick.</p>
<p>“Are you sure you should eat so fast?”</p>
<p>Erin asked Ryoka, as the second hamburger became crumbs before her eyes. Ryoka just blinked at Erin, and the girl threw up her hands.</p>
<p>“Fine. But if you throw up, you get to clean it up. Toren’s still missing, and I’m not doing it.”</p>
<p>“That’s fine. Can I get two more? Three?”</p>
<p>It was so <em>good</em> to eat food from her world! But Ryoka did slow down as Erin brought the third hot hamburger out. She was just ravenous, both from her run and her memories.</p>
<p>Val sat next to her, munching on some french fries Erin had made. He dipped one in ketchup and eyed Ryoka.</p>
<p>He really wasn’t what she’d expected. And the charm he’d lost because of her—</p>
<p>“Thanks again.”</p>
<p>He shrugged.</p>
<p>“It happens. Better that I found out my charm <em>can</em> be overloaded than get into trouble down the road. Besides, the mage that gave it to me will give me a discount—she had better since she promised it wouldn’t ever break. A few deliveries and I’ll be able to buy it again.”</p>
<p>“Really? How much do you make per delivery?”</p>
<p>Val rolled his eyes towards the ceiling as he munched on a fry.</p>
<p>“Forty gold coins? It can go over a hundred or two hundred if the area’s dangerous or if I need to get there quick.”</p>
<p>“<em>Forty—?</em>”</p>
<p>Ceria choked on a piece of potato and Olesm thumped her on the back until she spat it out. Val looked amused.</p>
<p>“It’s not <em>that</em> much. The nobility has enough gold to throw around, and any Gold-rank adventurer or higher earns a hundred times that much if they get lucky in a dungeon.”</p>
<p>“Money’s different where you come from, I guess. Here, gold coins are worth a <em>lot.</em>”</p>
<p>Erin said that as she refilled both Runners’ glasses. Olesm nodded, but Val shrugged.</p>
<p>“There’s a different kind of currency among high-level classes. Once you get past Level 30, you become one of…a thousand on the continent? Something like that. Your skills become in such high demand that you’ll earn more money than anyone else. And of course anyone with a class that involves managing land or selling or buying goods earns thousands of gold coins each month.”</p>
<p>Ryoka had to shake her head, although the parallel to her world wasn’t that far off. Erin sighed wistfully at the thought of all that money and then took a seat at Val’s table, smiling at him.</p>
<p>“Well, I’m happy you came here. Unlike <em>some people</em>, you pay for your food and you also helped Ryoka. So anytime you want to come back here for a night or to eat food, I’ll be happy to serve you.”</p>
<p>Val looked regretful.</p>
<p>“I doubt you’ll see me in the near future. I don’t often get deliveries this far south.”</p>
<p>“Really? But you’re a Courier. Don’t you go everywhere?”</p>
<p>“To an extent. But my area’s usually around First Landing. This is far, even for me. In truth, I’m surprised I was called on to do this delivery. Normally there’s another Runner who makes the run around here, but apparently he was busy due to some problem with the undead in the area.”</p>
<p>“Another Courier?”</p>
<p>That was the first Ryoka had heard of this. She wondered if it were a Drake or a Gnoll. Val nodded, and gestured to the window facing Liscor with one thumb.</p>
<p>“He’s a Beastkin named Hawk. I’ve met him a few times. He’s fast enough to get to First Landing and back. I don’t know why he didn’t take it, but apparently there’s been some trouble with the undead around here so he couldn’t.”</p>
<p>Hawk? Ryoka didn’t know the name either, but Erin did. Her eyes widened.</p>
<p>“Oh, I met him just yesterday! He’s so <em>cute!</em>”</p>
<p>Ryoka blinked at Erin. Cute?</p>
<p>Val laughed loudly, and nodded a few times.</p>
<p>“Hah! He is, isn’t he? Just don’t say that to his face. Hawk is even better at fighting than I am, and he gets touchy when us humans make fun of him.”</p>
<p>“But he is. I mean, he also eats meat. Isn’t that <em>weird?</em>”</p>
<p>“Why? He’s not like an animal rabbit, and don’t Drakes eat lizards?”</p>
<p>Val and Erin glanced over at Olesm. The Drake was trying to teach Ceria how to play chess, but when he heard that he looked around, an uncharacteristic scowl on his face.</p>
<p>“We’re <em>not lizards!</em>”</p>
<p>“Sorry!”</p>
<p>Val raised a hand and turned back to his conversation with Erin.</p>
<p>“In truth, I’m surprised I was needed at all. This place has two people with a Courier’s speed. One’s Hawk, the other I know of only by reputation. You haven’t heard of him, have you? He’s known as the Gecko by some of the Couriers who’ve met him.”</p>
<p>“Gecko? Like a lizard?”</p>
<p>Olesm’s shoulders hunched. Erin threw a rolled up cloth napkin at his head and it bounced off.</p>
<p>“I’m not talking about Drakes!”</p>
<p>She looked at Val and Ryoka and shook her head.</p>
<p>“I’ve never heard of someone named Gecko either. Weird. There’s all these famous people in Liscor that I don’t know about.”</p>
<p>“Well, perhaps he—or she—is only famous to Runners. I suppose other people wouldn’t care as much.”</p>
<p>“Hm. Well, you know, Ryoka’s a really fast Runner. I bet you she could be a Courier too!”</p>
<p>Ryoka and Val looked at Erin, and Ryoka felt her cheeks heating up. She opened her mouth, but Val cut in.</p>
<p>“A Courier needs a lot of experience. Maybe someday, but right now I’d still call her one of the fastest City Runners I’ve ever met. Which is even more impressive since she doesn’t have any levels.”</p>
<p>“Well—I guess. But Ryoka’s good! Can you give her a hand? Like, I dunno, talk to people about giving her a test or something?”</p>
<p>Val rubbed at his chin thoughtfully and nodded.</p>
<p>“If you’d like to try becoming a Courier, I could try and get some Runners Guilds to give her some long-distance requests. Ryoka, what do you think?”</p>
<p>Val looked at Ryoka. She blinked at the table, slowly.</p>
<p>“Mm.”</p>
<p>Both Val and Erin looked at Ryoka and realized the girl was falling asleep right on top of her food. Ryoka’s head was so full of static she barely heard the two changing seats. The stamina potion had worn off, and all she wanted to do…was…sleep…</p>
<p>Ryoka’s head came down. Her face <em>smushed</em> into the burger and she sat up quick. What time was it?</p>
<p>The sun was still out, and Val was still in the inn. But he was standing by the door, talking to Erin.</p>
<p>“This is a nice place, and your food is unique. I’ll do some shopping around the city and come back here tonight. Unless you’re expecting a crowd?”</p>
<p>Erin made a face.</p>
<p>“I <em>was</em>, but it turns out everyone in the city’s making hamburgers but me. Come on over and I’ll have tons of good food, okay?”</p>
<p>“I wouldn’t mind more of those fried potato things. Even if you’re only serving hamburgers, I’ll be back.”</p>
<p>“I’m going to make Philly cheese steaks!”</p>
<p>“Why does everything you make have the word ‘cheese’ in it? Cheeseburgers, cheese steaks…I mean, cheese isn’t the main part of it, so why…?”</p>
<p>Olesm scratched at his spines. He looked around for support and found none.</p>
<p>Val opened the door as he gave Erin a salute with two fingers, and then recoiled. Several Frost Faeries were hanging outside the door, laughing. Val eyed them warily.</p>
<p>“I hate those things. They plague me on runs and I can’t lose them no matter how fast I go.”</p>
<p>&nbsp;</p>
<p><span style="color:#8ae8ff;">“Hah! The slow manling thinks he can lose us? What a joke!”</span></p>
<p>&nbsp;</p>
<p>Val ducked as a snowball sailed through the inn’s doorway and knocked all of the chess pieces off the board Olesm and Ceria were using. Amid the Drake’s anguished cry, he ducked out of the inn, running fast towards Liscor with the faeries in hot pursuit.</p>
<p>Erin propped her hands on her hips and shook her head as she stared out the window.</p>
<p>“We’ve got to do something about those faeries. They’re really mean. You won’t believe what they did to me earlier, Ryoka. Ryoka?”</p>
<p>Ryoka’s head was lolling, but she jerked upright and looked at Erin.</p>
<p>“What?”</p>
<p>“…Nevermind. You know, I’ve got a lot of beds upstairs if you want to sleep. We can talk later.”</p>
<p>“—No. No, I’ll stay awake until tonight.”</p>
<p>If she slept now, Ryoka would <em>never</em> get to sleep tonight, and she didn’t fancy staying up for 24 hours in a row just to fix the problem. She scowled, and pinched herself hard.</p>
<p>“I’ll be fine. What are you going to do?”</p>
<p>Erin shrugged and stared at the position of the sun in the sky.</p>
<p>“I was planning on opening the inn all day, but since no one’s coming…I guess I’ll buy more food for tonight. I want to stock up, and Selys wants to go to the bathhouse together. Want to come? You can meet Krshia and wash…everything…off.”</p>
<p>That sounded good to Ryoka. Not the going to Liscor or meeting the Gnoll part, but a hot bath would be bliss. She nodded.</p>
<p>“Okay.”</p>
<p>She was still exhausted as she and Erin walked into the city, leaving Olesm, Ceria, and Pisces to mind the inn. Erin said she had faith in Ceria to keep the other two out of trouble, and it looked like they were all going to play chess instead. It turned out that [Mages] liked the game almost as much as [Tacticians].</p>
<p>Stamina potions. Ryoka knew it was only that which had allowed her to reach Liscor so fast, but she was determined never to overuse one again. She felt like her body was lead, and every few seconds her eyes would start crossing and go blurry unless she concentrated.</p>
<p>Talking was even harder, but Ryoka was curious why Val had come to Erin from so far away.</p>
<p>“You sent out a chess puzzle?”</p>
<p>“Yeah. It wasn’t a <em>super</em> hard one, but pretty hard. I guess it interested whoever sent the puzzle in the first place, because they sent me this expensive chess board. Why? Is that a bad idea?”</p>
<p>Erin looked at Ryoka as the other girl walked along, frowning at her feet.</p>
<p>“It could be. Not a good idea to give away secrets. Could be a…bad thing.”</p>
<p>Ryoka was having trouble thinking. She’d tried to explain the value of indiscretion to Erin before, but Erin just wasn’t as leery of giving away secrets as Ryoka was. Probably a personality thing.</p>
<p>“If you’re too good…it might be a problem.”</p>
<p>“Really? Being good at chess is a bad thing?”</p>
<p>It might be, if the person sending the chess board ever came looking for Erin and uncovered secrets about her. That was Ryoka’s paranoia speaking, but it was still a possibility.</p>
<p>Then again, given the thousands of other things that could go wrong, maybe she shouldn’t worry.</p>
<p>Ryoka followed Erin mechanically into the city, ignoring the route they took until Erin paused. Ryoka sniffed. A familiar scent was in the air—that of roasting meat. She looked around.</p>
<p>Market Street was full of shopkeepers and pedestrians. But in the winter chill, a third group of seller had appeared, and that was the food vendor. But unlike days previous, they weren’t selling a variety of dishes to the cold and hungry public, but one, very familiar food product.</p>
<p>Hamburgers.</p>
<p>Every Gnoll and Drake on the street was selling them as they cooked the meat over braziers, exactly as Erin had done on the first day. They were all in competition, but it seemed the people on the street were still hungry enough to give everyone some business.</p>
<p>Each vendor had a queue, and a unique lure. Some had added different varieties of cheese to their cheeseburger; others had their own blend of spices, or different kinds of meat. One enterprising Drake had even created a quadruple-patty hamburger that Ryoka would dearly have loved to try.</p>
<p>The Gnolls and Drakes paused as they spotted Erin down the street. They didn’t <em>quite</em> meet Erin’s eyes, but neither did they stop flipping and selling burgers. Ryoka stared around and inhaled the scents of grilled meat. Damn it. She was getting hungry again.</p>
<p>“Well, now I know where all of my customers went.”</p>
<p>Erin scowled and kicked at some snow on the ground.</p>
<p>“Let’s go find Krshia.”</p>
<p>As Ryoka stumbled after Erin down the street, Erin glared at the food vendors.</p>
<p>“Hey, is there anything I can do to them for stealing my stuff? Can I claim copyright or plagiarism or something?”</p>
<p>“Enforce copyright? Not unless you’ve got an army. Or a really big hammer.”</p>
<p>Ryoka scowled at the spinning ground. She saw Erin was opening her mouth – probably to ask what kind of hammer – when the girl turned and smiled.</p>
<p>“Krshia!”</p>
<p>The large Gnoll smiled as she came out of her stall to greet Erin. She sniffed at Ryoka, and the girl tried not to scowl directly at her.</p>
<p>“It is good to see you this day, Erin Solstice. And I notice you have brought your friend.”</p>
<p>“Yeah! Ryoka just came in yesterday. Oh, and you’ll never guess what happened this morning, Krshia! I met this Runner named Blah and he blah blah blah…”</p>
<p>Ryoka tuned out Erin’s voice as she tried to stay upright. Coming here had been a mistake. All she wanted to do was bathe fast and fall asleep. But Erin would talk to her friend, and then Ryoka would probably have to sit in the bathhouse for hours, <em>talking…</em></p>
<p>God. She couldn’t deal with this right now. Ryoka just wanted Erin to finish whatever she was doing, so she could go back and sleep.</p>
<p>Music. Music would help.</p>
<p>Ryoka felt at her pocket. Her iPhone was still there, nice and safe in its case. She pulled it out and turned it on. She needed music. Something to keep her awake. Something with a <em>beat.</em></p>
<p>She scrolled down the list of songs, frowning at the small letters. Ryoka had over ten thousand songs on her iPhone, and that was no joke. She had a 32 GB model, and since she’d also uninstalled almost every app besides a few essentials, her iPhone was just a glorified music player that could do magical conference calls.</p>
<p>Mind you, not all the songs were good, or even acceptable. Ryoka had far too many songs she’d downloaded in the indiscretions of youth and never bothered to delete. But finally she found one she liked. This one was a classic.</p>
<p>A heavy bass thumped in Ryoka’s ears with a steady drum beat. She waited, tapping one toe for Freddie Mercury. She grinned as the music swept her away.</p>
<p><em>“Ain’t no sound but the sound of his feet, machine guns ready to go.”</em></p>
<p>Another One Bites the Dust by Queen thundered in Ryoka’s ears as she turned the volume up to maximum and left it that way. She had no fear of annoying Erin as she talked with Krshia—one of the things Ryoka spent lots of money on was good headphones.</p>
<p>Freddie Mercury had a voice that spoke to Ryoka’s heart. She opened her eyes, smiling, and noticed something odd across the street. One of the Gnoll shoppers had stopped, and he was doing something odd. Ryoka frowned.</p>
<p>Was that Gnoll…bobbing his head? Yes. He was.</p>
<p>And over there. Krshia had stopped talking to Erin and she was tapping her foot. To the beat.</p>
<p><em>“Are you happy are you satisfied? How long can you stand the heat?”</em></p>
<p>Ryoka looked around. Every Gnoll in the street had suddenly stopped, to the bewilderment of the other Drakes and few humans. Yes. They were <em>all</em> nodding to the sound of the music.</p>
<p><em>Shit.</em></p>
<p>Ryoka stopped the song. Instantly, the other Gnolls looked around at her reproachfully.</p>
<p>Son of a…they could <em>hear</em> the music even from across the street. Of course. Ryoka wanted to slap herself. They were like <em>dogs</em>. Their hearing was far better than a human’s.</p>
<p>Krshia nudged Erin and pointed to Ryoka’s iPhone.</p>
<p>“Mm. Your friend, she has something very interesting, yes?”</p>
<p>Ryoka scowled, tucked the iPhone back in her pocket, and crossed her arms. She glared. Erin winced, and whispered to Krshia.</p>
<p>“She’s sort of <em>grumpy</em> because she’s tired.”</p>
<p>Ryoka heard that, and it only made her mood worse. Especially because it was true. She looked at Erin pointedly.</p>
<p>“Time to go?”</p>
<p>She said it like a question, but it wasn’t. Erin hesitated, but Krshia had already gathered her order. Apologizing, Erin left.</p>
<p>“Hey. What was wrong with that? Did they hear your music? Why’s that a bad thing?”</p>
<p>“Because they could realize that I have technology not from this world. And—and—”</p>
<p>Ryoka shook her head. Something about conspiracy theories and uncovering the truth? She needed to lie down.</p>
<p>“Let’s just go. You said there was a bathhouse?”</p>
<p>“This way. But uh, your iPhone—”</p>
<p>“I’ll leave it. With my clothes.”</p>
<p>“Right.”</p>
<p>&nbsp;</p>
<p>&#8212;-</p>
<p><strong> </strong></p>
<p>The bath did do wonders for Ryoka, enough so that she managed to apologize and even add a little to Erin and Selys’ gossip. Not much though; Ryoka just lay back in the warm waters and enjoyed being.</p>
<p>Unfortunately, due to some cosmic fate, it seemed that Erin’s mood was inversely proportional to Ryoka’s own. The instant they got back to her inn and found that no new guests had come by the entire day, the normally upbeat innkeeper lost her good cheer.</p>
<p>Erin stared around her empty inn and threw up her hands in disgust.</p>
<p>“Is <em>every</em> inn, pub, and tavern selling hamburgers tonight?”</p>
<p>Selys nodded as she shook snow off of a travelling cloak.</p>
<p>“Pretty much. Even Peslas is selling them in his inn.”</p>
<p>Erin slumped against a table and put her head down on the smooth wood surface. Ryoka exchanged glances with Selys. She felt she should say something encouraging, as a friend. Something helpful.</p>
<p>But what?</p>
<p>‘Shit happens. Get over it?’</p>
<p>‘At least it’s not raining?’</p>
<p>‘We all die some day, every one of us?’</p>
<p>…No. Ryoka decided not to say anything. Selys patted Erin on the shoulder, and looked around the inn for help.</p>
<p>Pisces just blinked at her, but Olesm piped up. With dreadful forced cheerfulness, he held up a chessboard.</p>
<p>“How about we all play some chess? I’m sure that will make us all feel much better. Ceria and Pisces really like the game as well!”</p>
<p>Ryoka volunteered to play, and Selys was stuck cheering people on since she had no real interest or skill at chess. As it transpired, she and Olesm were better players than the two mages, so both Ryoka and Olesm took on Erin.</p>
<p>Unfortunately, that turned out to be a mistake. Good as they were, even they weren’t close to a match for Erin. She beat them in brutal one-sided matches and then sulked at a table.</p>
<p>Ryoka gave up. She didn’t know how to cheer people up—she barely managed to stay happy herself most of the time. She pulled out her iPhone and stared at it. Then she threw caution to the wind and showed it to Pisces again.</p>
<p>“Can you cast [Repair] on this?”</p>
<p>The mage’s eyes flicked over Ryoka’s iPhone with unconcealed interest and avarice.</p>
<p>“I do know the spell. May I ask what’s wrong with this, ah, device?”</p>
<p>“It’s running out of…it needs mana, I guess you could say. Look, can you fix it?”</p>
<p>“If I do…would you permit me to inspect this object for a few minutes?”</p>
<p>“No. That won’t be possible.”</p>
<p>“Well, if that’s the case…”</p>
<p>Pisces shrugged as if there was nothing he could do. Ryoka ground her teeth, but then Erin spoke up.</p>
<p>“Do it. Or I’ll throw this at you.”</p>
<p>Erin didn’t raise her head, but she was holding a fork. Pisces hesitated. He was clearly thinking of Erin’s uncanny skill at throwing objects. He sniffed.</p>
<p>“Very well. [Repair].”</p>
<p>This time her iPhone didn’t change noticeably, but the screen flickered for just a second. When Ryoka looked again, the battery was at 100%. Perfect.</p>
<p>Ryoka flicked to her songs, but then she noticed Erin was staring at her phone. Right. Erin had told her that she’d left her phone back in her home when she was teleported here.</p>
<p>The look in Erin’s eyes was pure longing. She pointed hesitantly at the iPhone in Ryoka’s hands.</p>
<p>“Can I see it?”</p>
<p>Ryoka hesitated. Normally, her response to anyone touching her iPhone was to lay hands on them in an unfriendly way. She never lent it to anyone. But Erin—Erin was different.</p>
<p>Reluctantly, Ryoka handed the iPhone over.</p>
<p>“Here.”</p>
<p>She couldn’t even say ‘be careful with it’, because it could be repaired with a single spell. But it made Ryoka’s insides twist with embarrassment as she realized Erin was going through her songs.</p>
<p>“Oh my god. You’ve got <em>Pokémon </em>on here? That’s awesome!”</p>
<p>Her cheeks lit up. Ryoka was tempted to snatch the iPhone back, but it was too late. She gritted her teeth.</p>
<p>“It was a phase. I put lot of songs on there as a kid and a teen. I don’t listen to them anymore.”</p>
<p>That was a flat-out lie, but Ryoka would be dead and rotting before she admitted to listening to some of the songs she had on her iPhone. She’d gone through a lot of phases she wasn’t proud of.</p>
<p>“Oh man. This is just—”</p>
<p>At least Erin was happy. Ryoka saw her face light up, and realized she should have given Erin the iPhone a long time ago. She’d taken it for granted and listened to the music sometimes before she slept or as she ran, but Erin had nothing from their world. Ryoka took a seat as Erin closed her eyes and smiled.</p>
<p>Ceria eyed the earbuds in Erin’s ears with interest. They were sports gear, which meant they could stick in the ears even when Ryoka was running over rough terrain. They didn’t leak much sound, but Ryoka saw the half-Elf’s head tilt and her ears <em>twitch</em> ever so slightly.</p>
<p>“That’s so odd.”</p>
<p>Pisces was listening too. He had one finger touching his earlobe, and Ryoka saw a glowing blue aura around the tip of his finger and ear. She scowled, but both mages seemed to be having trouble listening to the faint audio that was making Erin smile.</p>
<p>Olesm and Selys looked at the mages and at Erin with clear interest, but they could hear nothing. Olesm prodded Erin gently.</p>
<p>“What is it? Can I hear?”</p>
<p>Erin looked startled, but then she nodded.</p>
<p>“Oh yeah, sure. This thing has speakers, right Ryoka?”</p>
<p>“That’s not a good—”</p>
<p>Ryoka trailed off and gave up. Ceria had already seen it once anyways. And besides…</p>
<p>“Sure. Yeah. Go ahead.”</p>
<p>Erin pulled the headphones out of the jack and the iPhone immediately began playing. Ryoka heard “Hey There Delilah” playing quietly in the silence.</p>
<p>The four guests in the inn froze. They each reacted differently.</p>
<p>Ceria’s eyes went wide as she heard the strange music she had listened to once before. Memory flashed in her eyes, and her eyes filled with tears and she took a seat. Ceria closed her eyes and remembered.</p>
<p>Olesm stared at the iPhone, mouth open, showing rows of sharp teeth. He listened with innocent wonder and his tail slowly began to wag to the sound of the gentle beat.</p>
<p>Selys blinked a few times, and then she took a seat, listening hard to the lyrics and closing her eyes.</p>
<p>Pisces—he stared out the window at the fading light in the sky without saying a word. He brushed at his eyes as the song ended.</p>
<p>And Erin and Ryoka listened and forgot for a moment they were in this world. Both girls felt pricking at their eyes. Erin wiped hers with her sleeve, and Ryoka just shook her head.</p>
<p>“Is there a way to make it louder? The volume’s maxed out, but…”</p>
<p>Erin trailed off as she clicked through the settings. Ryoka didn’t have a good solution besides buying a pair of speakers, but Pisces reached for the iPhone.</p>
<p>“A simple problem. Allow me.”</p>
<p>He tapped the metallic casing of the iPhone and whispered a spell.</p>
<p>“[Amplify Sound].”</p>
<p>Suddenly, the iPhone was loud enough to fill the entire room. Erin jumped, and then grinned as she stared down at the iPhone.</p>
<p>“Whoa. That’s <em>so</em> much better than speakers.”</p>
<p>Ryoka stared at her phone, suddenly very interested.</p>
<p>“How long does that spell last?”</p>
<p>It might be a terrible idea to play music while running in dangerous spots, but on the open road…Pisces raised one hand upwards and shrugged.</p>
<p>“I may be able to prolong the spell for several days with sufficient remuneration.”</p>
<p>“<em>I </em>can learn that spell in less than a day.”</p>
<p>Ceria frowned at Pisces and he scowled at her, but Erin still focused on the iPhone. She put a hand on her chin, and then <em>slapped</em> her forehead loud enough to make everyone look at her in concern.</p>
<p>Erin was suddenly excited. She halted the current song and held the iPhone out to Pisces.</p>
<p>“How loud <em>can</em> you make this?”</p>
<p>“How loud would you like it to be?”</p>
<p>“Really loud. Super loud! Loud enough to be heard even in the city.”</p>
<p>Pisces nodded, and frowned as he adjusted the spell, whispering words too quiet to hear. Ryoka eyed her friend.</p>
<p>“Erin. What are you doing?”</p>
<p>Erin turned and grinned at Ryoka.</p>
<p>“Getting customers!”</p>
<p>When she hit the play button on the iPhone, the sound that emerged nearly blew out everyone’s eardrums. Erin clapped her hands over her ears, but she was grinning wildly.</p>
<p>“Perfect! Now you all stay here and I’ll go get some business!”</p>
<p>She ran out of the inn. Ryoka followed her, and saw Erin climbing up to the roof. She shouted at Erin, but the other girl didn’t stop. Erin stood on top of the roof, and opened her mouth.</p>
<p>&nbsp;</p>
<p><span style="font-size:200%;">“Hey! It’s <em>music time! Let&#8217;s start with a classic!</em>”</span></p>
<p>&nbsp;</p>
<p>Ryoka clapped her ears to her head. Erin’s voice was <em>loud!</em> It had to be a Skill—it sounded like Erin was shouting into a megaphone. But then Erin touched the iPhone and Ryoka realized what she was about to do.</p>
<p>The drumbeats that started blasting through the air sounded like air cannons going off. But in a second Ryoka heard the guitar playing chords she knew by heart. Erin took a deep breath, and then she starting singing at the exact same time Kid Rock did.</p>
<p>Ryoka gaped at the ceiling as Erin began singing, perfectly matching the lyrics to the music. The only difference was that Erin changed the lyrics from boy to ‘girl’ and man to—a really fast ‘woman’, which Ryoka entirely approved of. Erin sang along with the song, projecting her voice out across the open plains and towards Liscor.</p>
<p>The sound <em>pulsed</em> through the inn, so loud that Ryoka could feel bass in her bones. She couldn’t wipe the grin off her face, especially when she heard Erin try to sing along with the female chorus.</p>
<p>&nbsp;</p>
<p><span style="font-size:200%;">“Whooaaaoh? Wooaaaoh!”</span></p>
<p>&nbsp;</p>
<p>Erin belted out the rest of the song until the last note. Ryoka expected her to stop, but Erin just switched to the next song. And then Ryoka heard something that made her blood chill.</p>
<p><em>No.</em></p>
<p>Thundering drums. The sounds of jungle wildlife. A cinematic orchestra, straight from the movie trailer Ryoka had ripped the song from.</p>
<p>No. She didn’t. But she did. Erin had the iPhone on <em>shuffle.</em> Ryoka started towards the door, but it was too late.</p>
<p>&nbsp;</p>
<p><span style="font-size:200%;"><em>“This one’s from </em>Tarzan<em>! How did the lyrics go? Wait, I remember! Here goes!”</em></span></p>
<p>&nbsp;</p>
<p>Ryoka listened in half-pain, half-amazement as Erin sang along with the song. How the hell did she remember the lyrics? But now Ryoka could practically see Tarzan on the screen in front of her.</p>
<p>As the song ended, Erin shouted down at Ryoka in a slightly-less loud voice.</p>
<p>“Oh hey, you’ve got <em>tons</em> of Disney songs on here!”</p>
<p>Ryoka covered her eyes with one hand. But she couldn’t stop grinning, even so. She remembered the songs right out of her childhood. Erin <em>clicked</em> through the songs, and then Ryoka heard a familiar drumbeat. Erin listened with a grin to the sounds of characters from Mulan falling down, and then her voice echoed off the city walls.</p>
<p>God. She even remembered the parts that <em>weren’t </em>part of the song.</p>
<p>It was amazing. Erin wasn’t a great singer, but she sang far better than Ryoka ever could. Ryoka had taken lessons from a professional along with a bevy of other lessons from tutors her parents hired, but she didn’t like singing. But even though Erin’s breathing was out of synch and she sometimes sang an entire octave off, the passion she put in her songs made up for all of it.</p>
<p>Ryoka stood in the kitchen as Erin sang a “I’ll Make a Man Out of You” from Mulan, and then ad-libbed to the wordless “Circle of Life” song from Lion King. And like the animals from Pride Rock, her words were the herald to a flood.</p>
<p>They came out of the city in ones and twos at first, and then the few became a stream which became a flood of people. Drakes, Gnolls, and humans too, following the sound of music.</p>
<p>They stood around Erin’s inn, listening to her as she belted out her world’s greatest hits filtered by Ryoka’s tastes. But there was the thing—hate the music or love it, Ryoka had a <em>lot</em> of the billboard’s top 100 songs across each passing year, and each one was preserved perfectly on her iPhone.</p>
<p>To the listeners, it was like hearing each classic the very first time. And this was a world that had never heard pop, rock, rap, or half of the musical styles evolved over a hundred years to the art forms they were today.</p>
<p>The audience was spellbound. Or rather, they were absorbed into the music, because the last thing they were doing was sitting still.</p>
<p>The Drakes and Gnolls were <em>dancing!</em> Ryoka saw what looked like classic ballroom dancing mixed with wild break dancing from a familiar Drake guardsman. The Gnolls had their own moves as well, and they moved to the beat in a way Ryoka could only call <em>primal.</em></p>
<p>“Oh my god. It’s a<em> mosh pit.</em>”</p>
<p>She couldn’t hear the sound of her own voice. Erin had somehow managed to turn <em>up</em> the volume of her song, and she was drowning out even the roar of the crowd below.</p>
<p>One of the people right below the inn was shouting up at Erin. Zevara had [Loud Voice] as a Skill as well, or she had a great set of lungs, but the iPhone combined with Erin’s voice was loud enough to drown the angry Drake Captain out.</p>
<p>Ryoka grinned as she heard Erin switch songs. She ducked into the inn and saw Ceria, Selys, and Olesm all scrambling to cook burgers as Erin sang her heart out on top of the inn.</p>
<p>&nbsp;</p>
<p><span style="font-size:200%;"><em>“This one’s called</em> Apologize! <em>And it&#8217;s directed at you, Zevara!</em>”</span></p>
<p>&nbsp;</p>
<p>Zevara tried to climb up the side of the inn, but other people in the crowd dragged her back. Erin countered by launching into “I Gotta Feeling”, and the rest was history.</p>
<p>Ryoka realized people were streaming into the inn as well. She made her way inside, and saw a huge press of bodies, held off at the kitchen by a wide-eyed Olesm. He was trying to keep order, and failing.</p>
<p>Somehow, Ryoka ducked through the crowd and shoved her way into the kitchen. She saw Ceria and Selys desperately trying to make Erin’s hamburgers. She’d left plenty of ground beef and ingredients, but the Drake and half-Elf were trying to make burgers and serve them to the hungry people outside while frying the meat. In Ceria’s case, one-handed.</p>
<p>“Here! Let me!”</p>
<p>Ryoka had to shout in Selys’ ear before the Drake surrendered the spatula. Ryoka began flipping burgers and dumping them onto a plate while Selys began adding buns and ingredients to the burger. She had no skills, but Ryoka knew how to cook.</p>
<p>The bass made the oil in the skillet <em>bounce</em>, but Ryoka started cooking as the beat blasted through the air. She flipped and served as fast as she could, losing herself to the music and the work. A scaled or fair-skinned hand was always there to take whatever Ryoka made.</p>
<p>She took her place serving food as well so Olesm could retreat and rest for a few moments. It wasn’t like taking orders though; people shoved money at her and grabbed food. Ryoka soon found herself covered in a mix of grease and food particles.</p>
<p>But they were selling everything and anything. Hamburgers were just the start; the people in the inn were demanding anything they could drink or consume, and Ryoka and the others struggled to meet the demands.</p>
<p>After they ran out of ground beef, they sent Olesm down to buy more and started cooking whatever Erin had in the kitchen. The crowd didn’t care; bowls of buttery noodles disappeared as fast as extra-crispy fried potato slices.</p>
<p>Olesm appeared thirty minutes later—an age—with welcome fresh supplies. Not only the ground beef; he’d bought everything he could in the market.</p>
<p>He’d also managed to badger Relc and Klbkch into helping Krshia roll up kegs of some kind of ale up to the inn. They nearly got mobbed by the crowd until Olesm set up one of the tables outside with the barrels and as many mugs as he could gather.</p>
<p>That opened the floodgates. Ryoka had thought people were flooding the inn before; now they couldn’t even see the floor. People crowded around, snatching mugs as money found its sticky way into a jar that had been filled with black flies. Klbkch took the flies.</p>
<p>Ryoka took only one moment to take a break. She staggered outside and saw that the inn wasn’t the only place doing business. Even as Erin sang on, more enterprising vendors had hauled out wood stands onto the grass and mud. The snow had long melted from the sheer body heat generated by the crowd.</p>
<p>Movement in the air made Ryoka look up. Above Erin as she stood on the inn, glowing blue shapes were flying through the dark night sky. Frost Faeries.</p>
<p>They were hovering in the air, dancing around, flying in complex patterns and laughing as Erin sang. For once, they seemed to be too caught up in the music to play any pranks. But one of them had…a mug? Ryoka saw the faerie empty the liquid and hurl the mug down into the crowd.</p>
<p>It shattered across the back of Relc’s head, but Ryoka didn’t think he or the people around him even noticed. They were caught up in the unfamiliar music, and the feeling in the air was pure excitement.</p>
<p>Erin started singing “All the Single Ladies” and half the crowd went wild. Ryoka had no idea Drakes even used rings, but then she saw a few on tails. And then she saw a few rings disappear and people start slipping off in pairs as the night progressed.</p>
<p>&nbsp;</p>
<p>&#8212;-</p>
<p>&nbsp;</p>
<p>How long did it last? Ryoka didn’t know, only that after a while even the adrenaline and music couldn’t keep her upright. She stumbled around, serving the waning crowd. At one point she saw Zevara shouting at Erin as the girl paused to take a drink and eat some food Ryoka passed up to her.</p>
<p>“People are trying to <em>sleep!</em> Stop singing!”</p>
<p>“<em>No!</em>”</p>
<p>It was a roar from the crowd below. Zevara looked around in disgust, and Erin tossed down the rest of her water. She stood up, and the <em>cheering</em> nearly deafened Ryoka.</p>
<p>“You can’t stop this! Yeah, you can’t stop me! Because—hit it!”</p>
<p>And of course she just had to sing “Can&#8217;t Stop The Feeling”. Ryoka had forgotten she <em>had</em> that song on her iPhone. Why? Oh, right, because it was catchy. Ryoka couldn’t help it. She moved her feet to the song. Just a bit.</p>
<p>…She might have danced with a Gnoll, and found herself competing to see who could perform more tricks with Relc. Ryoka knew how to do standing backflips; it turned out Relc could do that and break dance, but no one in the world knew how to moon walk.</p>
<p>For one glorious night Ryoka forgot about everything. She danced, laughed, and let the music sweep her thoughts away. She saw Erin laughing and dancing on the roof, and laughed too. The songs carried Ryoka up on a rush and carried her away until suddenly there was only darkness and silence.</p>
<p>From somewhere on top of the inn, Erin turned off the iPhone and stopped singing. The Drakes and Gnolls stumbled back towards the city, and Ryoka realized that the world was spinning around her.</p>
<p>Olesm and Selys had already staggered upstairs, leaving a trail in the trash strewn about on the floor. Pisces helped Ceria climb the stairs, and then wandered unsteadily out of the inn.</p>
<p>Ryoka looked at the stairs and didn’t even bother trying to ascend. She saw Val wave at her as he walked upstairs, waved back, and collapsed in a chair.</p>
<p>Her eyes closed. Ryoka was asleep before her head even fell back.</p>
<p>She slept, and dreamed of dragons.</p>
<p>&nbsp;</p>
<p>&#8212;-</p>
<p>&nbsp;</p>
<p>[Innkeeper Level 21!]</p>
<p>[Skill – Advanced Cooking Obtained!]</p>
<p>[Skill – Advanced Crafting Obtained!]</p>
<p>&nbsp;</p>
<p>[Singer Class Obtained!]</p>
<p>[Singer Level 6!]</p>
<p>[Skill – Perfect Recall Obtained!]</p>
<p>[Skill – Control Pitch Obtained!]</p>
<p>&nbsp;</p>
<p>Erin heard the words sometime right before she passed out. She woke up – still on top of the inn’s roof just past dawn, mainly because she was slipping off.</p>
<p>She nearly fell getting to the ground anyways, but managed to save herself just in time. Erin stared around the trashed landscape, and wondered if the snow would hide it all. No. She’d probably have to clean it up.</p>
<p>That went double for the inside of Erin’s inn. She took one look at the chaos and covered her eyes. But that wouldn’t help.</p>
<p>Without Toren, none of the dirty dishes and messy tables would miraculously clean themselves overnight. Erin stared around the trashed room and wished with all her heart that her skeleton would come back.</p>
<p>She <em>hated</em> cleaning up messes. Dust was what Erin was good at.</p>
<p>But something caught Erin’s eye as she stared around and tried to find a place to start cleaning. An oddity.</p>
<p>She <em>knew</em> no one had come by after the last patron had staggered out and Ceria had gone to sleep upstairs. But something had moved in the night.</p>
<p>Everything in her inn was the same as it had been when Erin had passed out. Except for one thing.</p>
<p>Erin stared at the chessboard Val had delivered to her, sitting at the far end of her bar. The rows of orderly pieces stared back at her, nice and uniform.</p>
<p>But—she’d never put any pieces on the chess board. And as Erin hesitantly stepped closer, she saw something else.</p>
<p>The pieces were transparent. Each one had a ghostly quality to it. They looked like a medieval set of chess, with vaguely human pieces in all the right places. But wisps rose from the pieces like smoke, and they had a faintly unearthly quality to their substance.</p>
<p>They looked like <em>magic</em>, and the coolest pieces Erin had ever seen in her life. And they were all perfectly set up in their starting positions on the board.</p>
<p>Except for one piece. A white pawn piece had moved up two squares, occupying one of the most traditional opening spots in the game. Pawn to E4. King’s Pawn Opening. Erin stared at it.</p>
<p>“Whoa.”</p>
